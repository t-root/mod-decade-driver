// Test DFPlayer trên ESP32-C3 Super Mini
// Kết nối:
//  - DFPlayer TX -> GPIO20 (RX1 của ESP32-C3)
//  - DFPlayer RX -> GPIO21 (TX1 của ESP32-C3)
//  - VCC, GND như bình thường
#include "HardwareSerial.h"
#include "DFRobotDFPlayerMini.h"

HardwareSerial mySerial(1);
DFRobotDFPlayerMini player;

bool playerReady = false;

void printMenu() {
  Serial.println("\n=== MENU ===");
  Serial.println("1 -> play 0001.mp3");
  Serial.println("2 -> play 0002.mp3");
  Serial.println("a -> /01/001.mp3");
  Serial.println("b -> /01/002.mp3");
  Serial.println("s -> stop");
  Serial.println("h -> menu");
}

void handleCommand(char cmd) {
  if (!playerReady) return;

  switch (cmd) {
    case '1': player.play(1); break;
    case '2': player.play(2); break;
    case 'a': player.playFolder(1, 1); break;
    case 'b': player.playFolder(1, 2); break;
    case 's': player.stop(); break;
    case 'h': printMenu(); break;
  }
}

void setup() {
  Serial.begin(115200);
  delay(1000);

  Serial.println("Khoi dong DFPlayer (ESP32-C3)...");

  // ===== ESP32-C3 UART1: RX=GPIO20, TX=GPIO21 =====
  Serial.println("Config: RX=20, TX=21 (UART1)");
  mySerial.begin(9600, SERIAL_8N1, 20, 21);
  delay(1500);

  if (player.begin(mySerial)) {
    Serial.println("DFPlayer OK");
    playerReady = true;

    player.volume(25);
    player.play(1);

    printMenu();
  } else {
    Serial.println("Khong tim thay DFPlayer!");
  }
}

void loop() {
  if (Serial.available()) {
    char cmd = tolower(Serial.read());
    handleCommand(cmd);
  }
}
